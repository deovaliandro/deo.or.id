Selection Sort

Ditulis oleh: Deo Valiandro. M <valiandrod@gmail.com>, pada 31 January 2017 
#include "stdio.h"

void swap(int a[], int i, int j)
{
    int tmp;
    tmp = a[j];
    a[j] = a[i];
    a[i] = tmp;
}

void selection_sort(int arr[])
{
    int i, j, min;
    for (i = 0; i < 4; i++) {
        min = i;
        for (j = i; j < 4; j++) {
            if (arr[j] < arr[min]) {
                min = j;
            }
        }
        swap(arr, i, min);
    }

    for (int k = 0; k < 4; k++) {
        printf("%d ", arr[k]);
    }
    printf("\n");
}

int main()
{
    int arr[] = {3, 4, 2, 1};
    selection_sort(arr);
}
